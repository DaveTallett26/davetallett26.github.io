MySQL IsNull Function
Translating MSAccess JET SQL into MySQL syntax this function saves a translation to IFNULL 

DELIMITER $$

DROP FUNCTION IF EXISTS `mydatabase`.`IsNull` $$
CREATE DEFINER=`root`@`localhost` FUNCTION `IsNull`(sIn varchar(1024)) RETURNS tinyint(1)
RETURN If(IFNULL(sIn,TRUE),TRUE,FALSE) $$

DELIMITER ;
